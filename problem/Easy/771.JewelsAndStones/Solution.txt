/*
*初步设想：
*	循环遍历S字符串的字符，再将得到的每一个字符与J字符串中的字符匹配，判断是否存在，如果存在则计数+1
*/
package msldxy.problem771;

class Solution {
    public int numJewelsInStones(String J, String S) {
        int count = 0;
        for(int i = 0;i < S.length(); i++){
            char ch = S.charAt(i);
            for(int j = 0;j < J.length(); j++){
                if(ch == J.charAt(j))
                    count++;
            }
        }
        return count;
    }
}
